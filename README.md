## MY PSEUDOCODE
// DATA STORAGE ( MOVIEDATA,OBJECT)
DECLARE 4 PARALLEL ARRAYS : TITLES,DIRECTOR,RATINGS,COMMENTS
FUN ADDMOVIE ( TITLE,DIRECTOR,RATING,COMMENT)
ADD DATA TO ALL 4 ARRAYS
END FUN

FUN CALCULATEAVERAGE(): SET TOTAL=0
LOOP FOR EACH RATING IN RATINGS :
TOTAL=TOTAL+RATING
RETURN TOTAL/RATINGS.SIZE
END FUN

INIT WITH 3 SAMPLE MOVIES BY CALLING ADDMOVIE()
//ADDMOVIEACTIVITY\
ON CREATE LINK ALL EDITTEXTS, BUTTONS,ERROR TEXTIEW
SET ONCLICK FOR BTNSAVE:
GET ALL TEXT FROM EDITTEXTS
CHECK FOR EMPTY FIELDS
IF TITLE,DIRECTOR,OR RATING IS EMPTY : SHOW ERROE "FIELDS REQUIRED"
STOP
\\DATA FOR MOVIE
MOVIEDATA.ADDMOVIE(TITLE,DIRECTORS.RARINGS,COMMENTS)
SHOW "MOVIE ADDED MESSAGE"
CLOSE ADDMOVIEACTIVITY
END

\\ DETAILEDVIEWSCREEN
LINK LISTVIEW , BUTTON,CREATE NEW EMPTY DISPLAYLIST 
LOOP FROM i=0 
GET TITLE,DIRECTOR,RATING,COMMENT AT INDEX i
CREATE formattedString = "$TITLE ($RATING/5) \nDirected by : $DIRECTOR \nComments: $COMMENT"
ADD formattedString TO DISPLAYLIST  
END LOOP
CREATE ADAPTER WITH DISPLAYLIST 
SET LISTVIEW ADAPTER
SET ONCLICK FOR BTNBACK
CLOSE
END
